# Calculando a porcentagem de cada valor em relação ao total para a coluna 'IN_ALIMENTACAO'

df_publicas = df[df['Tipo_Escola'] == 'Pública']
alimentacao = df_publicas.groupby('NO_MUNICIPIO')['IN_ALIMENTACAO'].value_counts(normalize=True).unstack().fillna(0)
porcentagem_alimentacao_0 = alimentacao[0] * 100  # Multiplicando por 100 para obter a porcentagem

# Selecionando os 20 maiores valores de porcentagem
top_20_porcentagem_alimentacao_0 = porcentagem_alimentacao_0.nlargest(20)

# Criando o gráfico de barras horizontais com as porcentagens
plt.figure(figsize=(10, 8))
barra = top_20_porcentagem_alimentacao_0.sort_values().plot(kind='barh', color='pink')
plt.title('Porcentagem de escolas públicas sem alimentação por município', fontweight='bold', fontsize=16)
plt.xlabel('Porcentagem (%)')
plt.ylabel('Município')
plt.xticks(fontsize=13)
plt.grid()
plt.tight_layout()
plt.show()

Esse gráfico exibe a porcentagem de escolas públicas por município que NÃO oferecem alimentação escolar para os alunos. Detalhes do gráfico: É um gráfico de barras, no eixo Y temos os 20 municípios com maiores porcentagens, no eixo X temos a porcentagem.
