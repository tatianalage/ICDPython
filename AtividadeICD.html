# Calculando a porcentagem de cada valor em relao ao total para a coluna 'IN_ALIMENTACAO'

df_publicas = df[df['Tipo_Escola'] == 'Pblica']
alimentacao = df_publicas.groupby('NO_MUNICIPIO')['IN_ALIMENTACAO'].value_counts(normalize=True).unstack().fillna(0)
porcentagem_alimentacao_0 = alimentacao[0] * 100  # Multiplicando por 100 para obter a porcentagem

# Selecionando os 20 maiores valores de porcentagem
top_20_porcentagem_alimentacao_0 = porcentagem_alimentacao_0.nlargest(20)

# Criando o grfico de barras horizontais com as porcentagens
plt.figure(figsize=(10, 8))
barra = top_20_porcentagem_alimentacao_0.sort_values().plot(kind='barh', color='pink')
plt.title('Porcentagem de escolas pblicas sem alimentao por municpio', fontweight='bold', fontsize=16)
plt.xlabel('Porcentagem (%)')
plt.ylabel('Municpio')
plt.xticks(fontsize=13)
plt.grid()
plt.tight_layout()
plt.show()

Esse grfico exibe a porcentagem de escolas pblicas por municpio que NO oferecem alimentao escolar para os alunos. Detalhes do grfico: um grfico de barras, no eixo Y temos os 20 municpios com maiores porcentagens, no eixo X temos a porcentagem.
