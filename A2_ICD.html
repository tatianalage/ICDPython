# Agrupa o Dataframe 'homicídios' por período
homicidios_anos = homicidios.groupby('período')
# Cria uma lista de estados únicos e a ordena
estados_lista = (sorted(list(set(homicidios['estado']))))

# Cria um novo dataframe vazio com colunas definidas pelos estados e índices
novo = pd.DataFrame(columns=[estados_lista], index=[''])

# Loop para os anos de 2000 a 2019
for i in range(2000,2020):

    # Cria um DataFrame vazio para armazenar as somas dos homicídios por estado
    somas = pd.DataFrame(columns=[estados_lista], index=[i])

    # Obtém o grupo correspondente ao ano atual
    atual = homicidios_anos.get_group(i)

    # Agrupa os dados do ano atual por estado e calcula a soma dos valores
    estados_soma = atual.groupby('estado').sum()

    # Preenche o DataFrame 'somas'
    for j in range(len(estados_lista)):
        somas[estados_lista[j]] = estados_soma['valor'][j]

    # Concatena o Dataframe 'somas' ao Dataframe 'novo'
    novo = pd.concat([novo, somas])

# Adiciona uma coluna 'Ano' com os valores dos índices (anos)
novo['Ano'] = novo.index.values

# Transforma o DataFrame 'novo' em um formato adequado para o gráfico de calor
Estados_reset = novo.melt(var_name='Sigla', value_name='Valor')

# Adiciona colunas 'index' e 'Ano' para controle posterior
Estados_reset['index'] = Estados_reset.index.values
Estados_reset['Ano'] = Estados_reset['index'].map(lambda x: (int(x) % 20) + 2000)

# Remove coluna 'index' e as últimas 20 linhas, que correspondem ao ano 2019
Estados_reset = Estados_reset.drop(['index'], axis=1).iloc[:-20]

# Cria um objeto de figura do Plotly
fig = go.Figure()

# Adiciona um trace inicial para o ano mínimo
heatmap = go.Choropleth(
    geojson="https://raw.githubusercontent.com/codeforamerica/click_that_hood/master/public/data/brazil-states.geojson",
    featureidkey="properties.sigla",
    locations=Estados_reset[Estados_reset['Ano'] == Estados_reset['Ano'].min()]['Sigla'],
    z=Estados_reset[Estados_reset['Ano'] == Estados_reset['Ano'].min()]['Valor'],
    colorscale="reds",
    colorbar=dict(title="Valores"),
)

fig.add_trace(heatmap)

# Adiciona um controle deslizante para escolher o ano
steps = []
for ano in Estados_reset['Ano'].unique():
    ano_data = Estados_reset[Estados_reset['Ano'] == ano]
    step = dict(
        args=[{'locations': [ano_data['Sigla'].tolist()], 'z': [ano_data['Valor'].tolist()]}],
        label=str(ano),
        method="restyle",
    )
    steps.append(step)

# Configura o controle deslizante
sliders = [dict(steps=steps, active=0, transition=dict(duration=300), x=0, y=0, len=1.0)]

# Atualiza o layout do mapa para melhor visualização
fig.update_geos(fitbounds="locations", visible=False, projection_type="mercator", bgcolor='lightblue')

# Ajusta a largura e altura do gráfico
fig.update_layout(width=1000, height=1000, sliders=sliders, title="Mapa de Calor Interativo - Valores por Estado (2000-2019)", title_x=0.5)

#Fazendo uma regressão para tentar aguma relação de registros e portes


from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score

# Separar o DF em conjuntos de treinamento e teste
train_df, test_df = train_test_split(merged_reg_morte, test_size=0.2, random_state=42)


X_train = train_df[['valor']]
y_train = train_df['TOTAL']

X_test = test_df[['valor']]
y_test = test_df['TOTAL']


model = LinearRegression()

# Treinar o modelo
model.fit(X_train, y_train)

# Fazer previsões no conjunto de teste
y_pred = model.predict(X_test)

# Avaliar o desempenho do modelo
mse = mean_squared_error(y_test, y_pred)
r2 = r2_score(y_test, y_pred)

# Mostrar os resultados
print(f'Mean Squared Error: {mse:.2f}')
print(f'R2 Score: {r2:.2f}')

# Coeficientes da regressão
print(f'\nCoefficients: {model.coef_}')
print(f'Intercept: {model.intercept_}')
